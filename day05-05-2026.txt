  ### Ansible - Create User and Set Permissions
     - name: Create a new user
       hosts: all
       become: yes
       tasks:
         - name: Create user
           user:
             name: pathnex
             state: present
             shell: /bin/bash
         - name: Set permissions for the user
           file:
             path: /home/pathnex
             owner: pathnex
             group: pathnex
             mode: '0755'
